# Plan: Modify Student Fees Page Column Structure

## Objective
Modify the existing **StudentFeePage** (`src/pages/StudentFeePage.tsx`) to display a table with these columns in exact order:

1. **Student** (name + enrollment number)
2. **Batch** (separate column)
3. **Enrolled Date** (admission date)
4. **Fees** (original fee)
5. **Discount**
6. **Total fees** (final fee = original - discount)
7. **Status** (badge)
8. **Pay** (action button)
9. **Receipt** (icon)
10. **Discount** (icon)
11. **Edit** (icon)
12. **Last Pay Date** (last payment date)

## Final Column Definition

```typescript
const studentColumns = [
  // 1. Student Name + Enrollment (without batch name)
  {
    key: "student_name",
    title: "Student",
    render: (fee: StudentFee) => (
      <div>
        <p className="text-sm font-semibold text-foreground">{fee.student_name}</p>
        <p className="text-[10px] text-muted-foreground uppercase font-medium">{fee.enrollment_no}</p>
      </div>
    ),
  },
  // 2. Batch (separate column)
  {
    key: "batch_name",
    title: "Batch",
    render: (fee: StudentFee) => (
      <span className="text-sm text-muted-foreground truncate block max-w-[150px]" title={fee.batch_name}>
        {fee.batch_name}
      </span>
    ),
  },
  // 3. Enrolled Date
  {
    key: "admission_date",
    title: "Enrolled Date",
    render: (fee: StudentFee) => (
      <span className="text-xs text-muted-foreground tabular-nums">
        {fee.admission_date ? new Date(fee.admission_date).toLocaleDateString() : "N/A"}
      </span>
    ),
  },
  // 4. Fees (Original Fee)
  {
    key: "original_fee",
    title: "Fees",
    render: (fee: StudentFee) => (
      <span className="text-sm tabular-nums">{formatCurrency(fee.original_fee)}</span>
    ),
  },
  // 5. Discount
  {
    key: "discount",
    title: "Discount",
    render: (fee: StudentFee) => (
      <span className={`text-sm tabular-nums ${fee.discount_amount > 0 ? "text-green-600" : "text-muted-foreground"}`}>
        {fee.discount_amount > 0 ? `-${formatCurrency(fee.discount_amount)}` : "None"}
      </span>
    ),
  },
  // 6. Total Fees (Final Fee)
  {
    key: "final_fee",
    title: "Total fees",
    render: (fee: StudentFee) => (
      <span className="text-sm font-bold text-foreground tabular-nums">{formatCurrency(fee.final_fee)}</span>
    ),
  },
  // 7. Status (Badge)
  {
    key: "status",
    title: "Status",
    render: (fee: StudentFee) => {
      const variant = feeStatusColors[fee.status];
      return <StatusBadge variant={variant}>{fee.status}</StatusBadge>;
    },
  },
  // 8. Pay Button
  {
    key: "pay",
    title: "Pay",
    render: (fee: StudentFee) => (
      <Button
        size="sm"
        variant="ghost"
        onClick={() => {
          setSelectedStudentFee(fee);
          setPaymentForm({
            studentFeeId: fee.id,
            paymentAmount: "",
            paymentMethod: "cash",
            paymentDate: new Date().toISOString().split("T")[0],
          });
          setAddPaymentOpen(true);
        }}
        className="h-7 text-xs"
        disabled={fee.final_fee <= fee.paid_fees}
      >
        Pay
      </Button>
    ),
  },
  // 9. Receipt Icon Button
  {
    key: "receipt",
    title: "",
    render: (fee: StudentFee) => (
      <Button
        size="sm"
        variant="ghost"
        onClick={() => generateFeeReceiptPDF(fee)}
        className="h-7 w-7 p-0"
        disabled={fee.paid_fees === 0}
        title="Generate Receipt"
      >
        <Printer className="w-3 h-3" />
      </Button>
    ),
  },
  // 10. Discount Icon Button
  {
    key: "discount_action",
    title: "",
    render: (fee: StudentFee) => (
      <Button
        size="sm"
        variant="ghost"
        onClick={() => {
          setSelectedStudentFee(fee);
          setDiscountForm({
            studentFeeId: fee.id,
            discountAmount: fee.discount_amount.toString(),
            discountReason: fee.discount_reason || "",
          });
          setAddDiscountOpen(true);
        }}
        className="h-7 w-7 p-0"
        title="Apply Discount"
      >
        <Percent className="w-3 h-3" />
      </Button>
    ),
  },
  // 11. Edit Icon Button
  {
    key: "edit",
    title: "",
    render: (fee: StudentFee) => (
      <Button
        size="sm"
        variant="ghost"
        onClick={() => {
          setSelectedStudentFee(fee);
          setStudentFeeForm({
            studentId: fee.student_id,
            batchFeeId: fee.batch_fee_id,
            originalFee: fee.original_fee.toString(),
            discountAmount: fee.discount_amount.toString(),
            discountReason: fee.discount_reason || "",
            status: fee.status,
          });
          setEditStudentFeeOpen(true);
        }}
        className="h-7 w-7 p-0"
        title="Edit Fee"
      >
        <Pencil className="w-3 h-3" />
      </Button>
    ),
  },
  // 12. Last Pay Date
  {
    key: "last_payment_date",
    title: "Last Pay Date",
    render: (fee: StudentFee) => (
      <span className="text-xs text-muted-foreground tabular-nums">
        {fee.last_payment_date ? new Date(fee.last_payment_date).toLocaleDateString() : "N/A"}
      </span>
    ),
  },
];
```
